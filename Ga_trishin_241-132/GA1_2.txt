#include <iostream>
#include <vector>
using namespace std;

int main()
{
    setlocale(LC_ALL, "rus");
    cout << "Введите кол-во символов в массиве" << endl;
    int b;
    cin >> b;
    vector<int> m;
    int c;
    cout << "Теперь введите символы" << endl;
    for (int i = 0; i < b; i++) {
        cin >> c;
        m.push_back(c);
    }
    int max;
    for (int j = 0; j < b; j++) {
        max = j;
        for (int x = j; x < b; x++) {
            if (m[x] > m[max]) {
                max = x;
            }
        }
        swap(m[j], m[max]);
    }
    cout << "сортировка: ";
    for (int i = 0; i < b; i++) {
        cout << m[i] << " ";
    }
}
