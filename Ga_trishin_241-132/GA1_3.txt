#include <iostream>
#include <vector>
using namespace std;


int main()
{
    setlocale(LC_ALL, "rus");
    cout << "Введите кол-во символов в массиве" << endl;
    int b;
    cin >> b;
    vector<int> m;
    string c;
    string num;
    int z;
    int p1 = pow(10, 2);
    int p2 = pow(10, 4);
    int p3;
    cout << "Теперь введите символы" << endl;
    for (int i = 0; i < b; i++) {
        cin >> c;
        num = "";
        for (int l = 0; l < 8; l++) {
            if (l % 3 != 2) {
                num += c[l];
            }
        }
        z = atof(num.c_str());
        m.push_back(z);
    }
    int min;
    for (int j = 0; j < b; j++) {
        min = j;
        for (int x = j; x < b; x++) {
            if (m[x] < m[min]) {
                min = x;
            }
        }
        swap(m[j], m[min]);
    }
    
    cout << "сортировка: ";
    for (int i = 0; i < b; i++) {
        p3 = (m[i] / p2) % p1;
        cout << m[i] / p2 << "-" << p3 << "-" << m[i] % p1 << endl;
    }
}
