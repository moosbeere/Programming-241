#include <iostream>
#include <locale>
using namespace std;

int main() {
    // Настройка локализации для русских букв
    setlocale(LC_ALL, "Russian");

    // 1. Запрос ввода положительного числа с проверкой
    int number;
    while (true) {
        cout << "Введите целое положительное число: ";
        cin >> number;

        if (number > 0) {
            break;
        }
        else {
            cout << "Ошибка! Число должно быть положительным. Попробуйте снова." << endl;
        }
    }

    // 2. Вычисление суммы чисел от 1 до введённого числа
    int sum = 0;
    cout << "Сумма чисел от 1 до " << number << ": ";
    for (int i = 1; i <= number; i++) {
        sum += i;
        cout << i;
        if (i < number) {
            cout << " + ";
        }
    }
    cout << " = " << sum << endl;

    // 3. Создание массива из 10 чисел
    int arr[10] = { 2, 5, 8, 12, 15, 18, 22, 25, 30, 35 };

    // 4. Вывод всех 10 чисел массива
    cout << "Массив из 10 чисел: ";
    for (int i = 0; i < 10; i++) {
        cout << arr[i];
        if (i < 9) {
            cout << " ";
        }
    }
    cout << endl;

    // 5. Вывод чисел на чётных позициях (используя %)
    cout << "Числа на чётных позициях: ";
    for (int i = 0; i < 10; i++) {
        if (i % 2 == 0) { // позиции 0, 2, 4, 6, 8
            cout << arr[i] << " ";
        }
    }
    cout << endl;

    // 6. Сумма элементов на нечётных позициях (используя %)
    int oddPosSum = 0;
    for (int i = 0; i < 10; i++) {
        if (i % 2 != 0) { // позиции 1, 3, 5, 7, 9
            oddPosSum += arr[i];
        }
    }
    cout << "Сумма элементов на нечётных позициях: " << oddPosSum << endl;

    return 0;
}
