#include <iostream>
using namespace std;

int sumOfDigits(int n) {
    if (n == 0) return 0;
    return (n % 10) + sumOfDigits(n / 10);
}

int main() {
    int n;
    cout << "Введите натуральное число N: ";
    cin >> n;
    
    cout << "Сумма цифр числа " << n << ": " << sumOfDigits(n) << endl;
    
    return 0;
}
