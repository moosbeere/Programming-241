#include <iostream>
using namespace std;

void printNumbers(int n) {
    if (n == 0) return;
    printNumbers(n - 1);
    cout << n << " ";
}

int main() {
    int n;
    cout << "Введите натуральное число n: ";
    cin >> n;
    
    cout << "Числа от 1 до " << n << ": ";
    printNumbers(n);
    cout << endl;
    
    return 0;
}
