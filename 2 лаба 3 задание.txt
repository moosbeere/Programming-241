// Функция из задания 10-11
int sumToN(int n = 1) {
    if (n <= 0) return 0;

    int sum = 0;
    for (int i = 1; i <= n; i++) {
        sum += i;
    }
    return sum;
}

// Функция из задания 12-13
void calculateSumAndProduct(int a, int b, int& sumRef, int& productRef) {
    sumRef = a + b;
    productRef = a * b;
}
